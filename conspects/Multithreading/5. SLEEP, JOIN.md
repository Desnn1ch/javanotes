
## JOIN

```java
public final class Runner {
    public static void main(String[] args) throws InterruptedException {
        // Создаём задачи
        TaskSummingNumbers task1 = new TaskSummingNumbers(1, 500);
        TaskSummingNumbers task2 = new TaskSummingNumbers(501, 1000);

        // Создаём потоки
        Thread thread1 = new Thread(task1, "Thread-1");
        Thread thread2 = new Thread(task2, "Thread-2");

        // Запускаем
        thread1.start();
        thread2.start();

        // Ждём завершения обоих потоков
        thread1.join(); // приостанавливает поток main пока thread1 не кончится
        thread2.join(); // приостанавливает поток main пока thread2 не кончится
        // Или можно вот так
        Thread.sleep(1000); // очевидно что это на глаз чисто, не факт что оба 
						    // потока закончат работу

        // Складываем результаты
        long resultNumber = task1.getResult() + task2.getResult();

        System.out.println("Сумма чисел от 1 до 1000 = " + resultNumber);
    }
}

class TaskSummingNumbers implements Runnable {
    private final int start;
    private final int end;
    private long result;

    public TaskSummingNumbers(int start, int end) {
        this.start = start;
        this.end = end;
    }

    @Override
    public void run() {
        long sum = 0;
        for (int i = start; i <= end; i++) {
            sum += i;
        }
        result = sum;
        System.out.println(Thread.currentThread().getName() +
                           " посчитал сумму в диапазоне [" + start + ";" + end + "] = " + result);
    }

    public long getResult() {
        return result;
    }
}
```
